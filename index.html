rawData = FileAttachment("data/consolidated/air_quality_consolidated.csv").csv({typed: true})

// Parsear fechas, calcular categorías y extraer métricas
data = rawData.map(d => {
  const dateObj = new Date(d.fecha_hora_registro);
  const ica = +d.valor_ica;
  const lat = +d.latitude;
  const lon = +d.longitude;
  
  let cat = "Desconocido";
  let color = "#999999";
  if (ica <= 50) { cat = "Buena"; color = "#00e400"; }
  else if (ica <= 100) { cat = "Moderada"; color = "#ffff00"; }
  else if (ica <= 150) { cat = "Dañina a Grupos Sensibles"; color = "#ff7e00"; }
  else if (ica <= 200) { cat = "Dañina a la Salud"; color = "#ff0000"; }
  else if (ica <= 300) { cat = "Muy Dañina"; color = "#8f3f97"; }
  else { cat = "Peligrosa"; color = "#7e0023"; }

  return {
    ...d,
    date: dateObj,
    hour: dateObj.getHours(),
    ica: ica,
    latitude: lat,
    longitude: lon,
    // Diferenciar lugares combinando el nombre con 3 dígitos de sus coordenadas
    lugar_nombre: `${d.lugar_nombre} [${lat.toFixed(3)}, ${lon.toFixed(3)}]`,
    categoria: cat,
    color: color
  };
}).filter(d => !isNaN(d.ica) && d.ica > 0 && d.ica <= 500 && !isNaN(d.latitude) && !isNaN(d.longitude))
  .sort((a, b) => d3.ascending(a.date, b.date));

lugares = Array.from(new Set(data.map(d => d.lugar_nombre))).sort()

minDateStr = d3.min(data, d => d.date).toISOString().split('T')[0]
maxDateStr = d3.max(data, d => d.date).toISOString().split('T')[0]

viewof selectedLugar = Inputs.select(["Todos"].concat(lugares), {label: "Ubicación:", value: "Todos"})
viewof startDateStr = Inputs.date({label: "Desde:", value: minDateStr})
viewof endDateStr = Inputs.date({label: "Hasta:", value: maxDateStr})
viewof startHour = Inputs.range([0, 23], {label: "Hora Mín.", step: 1, value: 0})
viewof endHour = Inputs.range([0, 23], {label: "Hora Máx.", step: 1, value: 23})
filteredData = data.filter(d => {
  const locMatch = selectedLugar === "Todos" || d.lugar_nombre === selectedLugar;
  const hourMatch = d.hour >= startHour && d.hour <= endHour;
  
  let dateMatch = true;
  if (startDateStr) {
    const sDate = new Date(startDateStr);
    dateMatch = dateMatch && (d.date >= sDate);
  }
  if (endDateStr) {
    const eDate = new Date(endDateStr);
    eDate.setHours(23, 59, 59);
    dateMatch = dateMatch && (d.date <= eDate);
  }
  
  return locMatch && hourMatch && dateMatch;
})

avgIca = filteredData.length ? d3.mean(filteredData, d => d.ica) : 0
maxIca = filteredData.length ? d3.max(filteredData, d => d.ica) : 0
